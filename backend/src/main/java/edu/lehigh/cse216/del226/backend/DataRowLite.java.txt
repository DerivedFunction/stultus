package edu.lehigh.cse216.del226.backend;

/**
 * DataRowLite holds a row of info, consisting of an id, and title.
 * Used to send data in JSON format, simple format.
 */
public class DataRowLite {
  /**
   * The id of the element (unchanged)
   */
  public final int mId;

  /**
   * The title  of the element
   */
  public String mTitle;

  /**
   * Create a new DataRowLite with data from DataRow
   * @param id
   * @param title
   * */
  DataRowLite(DataRow data) {
    this.mId = data.mId;
    this.mTitle = data.mTitle;
  }
  @Override
    public boolean equals(Object other) {
      DataRowLite obj = (DataRowLite)other;
      if (((Integer)this.mId).equals(obj.mId))
        return false;
      return this.mTitle.equals(obj.mTitle);
    }
}
